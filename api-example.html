Example 1: Get All Data

Retrieve all items from the civil and commercial code.

// Fetch JSON data directly
fetch('api/civil_and_commercial_code.json')
    .then(res => res.json())
    .then(data => {
        const items = data.civil_and_commercial_code;
        console.log({ code: 'civil_and_commercial_code', total: items.length, data: items });
    });

Example 2: Filter by ID

Get a specific law item by its ID.

// Fetch JSON and filter by ID
fetch('api/civil_and_commercial_code.json')
    .then(res => res.json())
    .then(data => {
        const items = data.civil_and_commercial_code.filter(item => 
            item.id == '1012' || item.id === 'มาตรา 1012'
        );
        console.log({ code: 'civil_and_commercial_code', total: items.length, data: items });
    });

Example 3: Search

Search for items containing specific text across all fields.

// Fetch JSON and search
const searchTerm = 'ห้างหุ้นส่วน';
fetch('api/civil_and_commercial_code.json')
    .then(res => res.json())
    .then(data => {
        const items = data.civil_and_commercial_code.filter(item =>
            item.title?.toLowerCase().includes(searchTerm.toLowerCase()) ||
            JSON.stringify(item.content).toLowerCase().includes(searchTerm.toLowerCase())
        );
        console.log({ code: 'civil_and_commercial_code', total: items.length, data: items });
    });

Example 4: Sort Descending

Get all items sorted by ID in descending order.

// Fetch JSON and sort descending
fetch('api/civil_and_commercial_code.json')
    .then(res => res.json())
    .then(data => {
        const items = data.civil_and_commercial_code.sort((a, b) => 
            String(b.id).localeCompare(String(a.id))
        );
        console.log({ code: 'civil_and_commercial_code', total: items.length, data: items });
    });

Example 5: Pagination

Get the first 5 items with pagination.

// Fetch JSON and paginate
fetch('api/civil_and_commercial_code.json')
    .then(res => res.json())
    .then(data => {
        const allItems = data.civil_and_commercial_code;
        const items = allItems.slice(0, 1);  // limit=1, offset=0
        console.log({ code: 'civil_and_commercial_code', total: allItems.length, 
                      offset: 0, limit: 1, count: items.length, data: items });
    });

Example 6: Complex Query

Combine multiple parameters: filter by title, sort, and limit results.

// Fetch JSON, filter, sort and limit
const title = 'สัญญา';
fetch('api/civil_and_commercial_code.json')
    .then(res => res.json())
    .then(data => {
        let items = data.civil_and_commercial_code.filter(item =>
            item.title?.toLowerCase().includes(title.toLowerCase())
        ).sort((a, b) => 
            String(a.id).localeCompare(String(b.id))
        ).slice(0, 5);
        console.log({ code: 'civil_and_commercial_code', total: items.length, data: items });
    });
